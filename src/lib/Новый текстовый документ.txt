// src/app/login/page.tsx
'use client'

import { useState } from 'react'
import { supabase } from '@/lib/supabase'
import { useRouter } from 'next/navigation'

export default function Login() {
  const [username, setUsername] = useState('')
  const [password, setPassword] = useState('')
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState('')
  const router = useRouter()

  const handleLogin = async () => {
    if (!username || !password) {
      setError('Ник и пароль — обязательно, братан')
      return
    }

    setLoading(true)
    setError('')

    const { data, error } = await supabase.auth.signInWithPassword({
      email: `${username}@krinzh.fake`,
      password,
    })

    if (error && error.message.includes('Invalid login credentials')) {
      const { error: signUpError } = await supabase.auth.signUp({
        email: `${username}@krinzh.fake`,
        password,
        options: { data: { username } }
      })

      if (signUpError) {
        setError('Что-то по пизде. Попробуй другой ник.')
      } else {
        router.push('/lobby')
      }
    } else if (error) {
      setError('Ошибка входа.')
    } else {
      router.push('/lobby')
    }

    setLoading(false)
  }

  return (
    <div className="min-h-screen bg-gradient-to-br from-purple-900 via-black to-red-900 flex items-center justify-center p-8">
      <div className="w-full max-w-2xl">
        <h1 className="text-9xl font-black text-center mb-4 text-transparent bg-clip-text bg-gradient-to-r from-yellow-400 via-pink-500 to-purple-600 animate-pulse">
          КРИНЖ ДО СМЕРТИ
        </h1>
        <p className="text-4xl text-center font-bold text-krinzh-300 mb-16">
          Вход в легенду
        </p>

        <div className="bg-zinc-900 bg-opacity-90 backdrop-blur-xl rounded-3xl p-12 border-4 border-krinzh-500 shadow-2xl">
          {error && (
            <div className="bg-red-900 border-2 border-red-600 text-red-200 px-8 py-6 rounded-2xl mb-8 text-2xl font-bold text-center animate-pulse">
              {error}
            </div>
          )}

          <input
            type="text"
            placeholder="Твой кринжовый ник"
            value={username}
            onChange={(e) => setUsername(e.target.value)}
            className="w-full px-10 py-8 mb-8 bg-black border-4 border-zinc-800 rounded-2xl text-3xl text-white placeholder-zinc-600 focus:border-krinzh-500 focus:outline-none transition-all"
            disabled={loading}
            autoFocus
          />

          <input
            type="password"
            placeholder="Пароль (любой)"
            value={password}
            onChange={(e) => setPassword(e.target.value)}
            onKeyDown={(e) => e.key === 'Enter' && handleLogin()}
            className="w-full px-10 py-8 mb-12 bg-black border-4 border-zinc-800 rounded-2xl text-3xl text-white placeholder-zinc-600 focus:border-krinzh-500 focus:outline-none transition-all"
            disabled={loading}
          />

          <button
            onClick={handleLogin}
            disabled={loading}
            className="w-full py-10 bg-gradient-to-r from-krinzh-600 via-purple-600 to-pink-600 text-6xl font-black rounded-3xl shadow-2xl hover:scale-105 hover:rotate-2 transition-all duration-300 disabled:opacity-50 border-8 border-yellow-500"
          >
            {loading ? 'КРИНЖИМ...' : 'ВОЙТИ В БОЙ'}
          </button>
        </div>
      </div>
    </div>
  )
}
